from qgis.core import *
from qgis.gui import *
import math

@qgsfunction(group='Custom', referenced_columns=[])
def prec2(polly):
    centr = polly.centroid().asPoint()
    sqmaxdist = 0
    i =0
    for v in polly.vertices():
        sqdist = polly.sqrDistToVertexAt(centr,i)
        if sqdist>sqmaxdist:
            sqmaxdist = sqdist
        i = i+1
    return math.sqrt(sqmaxdist)
